Реализуйте и экспортируйте по умолчанию функцию, которая возвращает первый результат, который не был ошибкой. Она должна запускать колбек на выполнение сразу для всех входящих параметров, а не последовательно. Это значительно увеличивает скорость ее выполнения.

### Примеры
    
    
    async.detect(['file1','file2','file3'], (filePath, callback) => {
      fs.access(filePath, err => {
        callback(null, !err)
      });
    }, (err, result) => {
        // result now equals the first file in the list that exists
    });
