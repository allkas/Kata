## src/main/java/io/hexlet/App.java

Реализуйте публичный статический метод `chunk()`, который принимает на вход массив строк и число, задающее размер чанка (куска). Метод должен вернуть двумерный массив, состоящий из чанков указанной размерности.
    
    
    String[] words = {"one", "two", "three", "four"};
    
    var chunks1 = App.chunk(words, 2);
    System.out.println(Arrays.deepToString(chunks1));
    // => [["one", "two"], ["three", "four"]]
    
    var chunks2 = App.chunk(words, 3);
    System.out.println(Arrays.deepToString(chunks2));
    // => [["one", "two", "three"], ["four"]]
